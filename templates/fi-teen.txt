# Valtteri fi-teen language

# Multi-message template example
#
# The first line after the template body relates to the first fact,
# The second line after the template relates to the second fact etc.
# Note how the second fact has a requirement for the facts to share a “who”

#Kokoomus sai eniten ääniä 1632 äänellään
#{1.who} sai eniten ääniä {2.what} äänellään
#| what_type = total_votes_rank, what = 1
#| what_type = total_votes, who = 1.who, where = 1.where


# Note that many of the templates below (though not all) use the terms “value”, 
# “name” and “place”, which are equivalent to “what”, “who” and “where”

# ---Total votes templates—

fi-teen: {name} sai [{place, case=ssa}] niinku {value} ääntä
fi-teen-head: {name, case=lle} {place, case=ssa} jotain {value} ääntä
| total_votes > 0

# extra special case
fi-teen: {name} sai [{place, case=n}] huudeilla jotain {value} ääntä 
| total_votes > 0, place_type = pollingstation

fi-teen: {name} feilas tosi pahasti, ne ei saanu [{place, case=ssa}] yhtään ääntä
#headline below
| total_votes = 0, name_type = party

fi-teen: eiks {name, case=lla} oo yhtään frendii koko kylässä, eiks edes sen mutsi asu {place, case=ssa} et se olis voinu äänestää sitä
#headline below
| total_votes = 0, name_type = candidate, place_type = municipality

fi-teen: {place} ignooras {name, case=n} kokonaan, eiks ne tienny et se on ehdolla
| total_votes = 0, name_type = candidate, place_type = pollingstation

# headline for all the above
fi-teen-head: OMG, {place} ignooras {name, case=n}
| total_votes = 0


# Total votes in previous election

fi-teen: {name} sai [{place, case=ssa}] viime vaaleissa {value} ääntä tai jotain
fi-teen-head: tb: {name, case=lle} viime vaaleissa {place, case=ssa} tota {value} ääntä
| total_votes_previous_election > 0

fi-teen: {place} ei antanu {name, case=lle} viimeks yhtään ääntä
fi-teen-head: {place} dissas {name, case=n} viimeks
| total_votes_previous_election = 0


# ---Percentage_votes templates---

fi-teen: {name} sai [{place, case=ssa}] {value, dot-to-comma} prossaa tai promillee tai desilitraa tai jotain
fi-teen-head: {name, case=lle} {place, case=ssa}] {value, dot-to-comma} prossaa äänistä
| value_type = percentage_votes

fi-teen: {name} sai [{place, case=ssa}] {value, dot-to-comma} prossaa äänist
fi-teen-head: {place} oli {name, case=lle} wörtti, vei yli kolmanneksen äänist
| percentage_votes > 34

fi-teen: {name} sai [{place, case=ssa}] siis jotain {value, dot-to-comma} prossaa äänist
fi-teen-head: ihquu, {place} anto yli puolet äänistä {name, case=lle}
| percentage_votes > 50.

fi-teen: {name} sai [{place, case=ssa}] {value, dot-to-comma} prossaa lisää niinku viime vaaleihin verrattuna
fi-teen-head: {name} on {place, case=lle} rks: {value, dot-to-comma} prossaa uusii äänii
| percentage_votes_change > 0

fi-teen: {name} sai [{place, case=ssa}] {value, abs, dot-to-comma} prosenttiyksikköö vähemmä äänii ku viime vaaleissa
fi-teen-head: {name, case=n} oreot pipariks {place, case=ssa}
| percentage_votes_change < 0

fi-teen: {name} sai [{place, case=ssa}] yhtä monta prossaa äänistä ku viimeks
fi-teen-head: tylsää: {name, case=n} kannatus ei muuttunut {place, case=ssa}
| percentage_votes_change = 0

# --Is candidate elected or in reserve (in this election)--
# fact election_result (exists for candidates and municipalities only?) has three possible values:
# election_result = 1: elected
# election_result = 2: in reserve
# election_result = 3: not elected

fi-teen: {name} valittii [{place, case=n}] valtuustoo
fi-teen-head: {name} salee {place, case=n} valtuustoo
| election_result = 1

fi-teen: {name} valittii varavaltuutetuks [{place, case=n}] valtuustoo
fi-teen-head: {name} varamiespalveluun {place, case=ssa}
| election_result = 2

fi-teen: {name} ei päässy [{place, case=n}] valtuustoo
fi-teen-head: {name} ei {place, case=n} valtuustoo
| election_result = 3


# In combination with previous results

fi-teen: {name} nousi [{place, case=n}] valtuustoo
fi-teen-head: {name} pääsee uutena {place, case=n} valtuustoo
| election_result = 1
| who = 1.who, where = 1.where, is_councillor = False

fi-teen: {name} jatkaa ton [{place, case=n}] valtuustos

| election_result = 1
| who = 1.who, where = 1.where, is_councillor = True

fi-teen: {name} ei päässy jatkamaa [{place, case=n}] valtuustost

| election_result = 3
| who = 1.who, where = 1.where, is_councillor = True
